class Solution {
    public int[] getConcatenation(int[] nums) {
        // Get the length of the input array
        int numsLength = nums.length;
        
        // Create a new array of size double the input array length
        int[] ans = new int[numsLength * 2];
        
        // Copy the elements of the input array into the first half of the new array
        System.arraycopy(nums, 0, ans, 0, numsLength);
        
        // Copy the elements of the input array into the second half of the new array
        System.arraycopy(nums, 0, ans, numsLength, numsLength);
        
        // Return the concatenated array
        return ans;
    }
}
